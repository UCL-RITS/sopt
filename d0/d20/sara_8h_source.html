151 void SARA::indirect(Eigen::ArrayBase<T1> const &coeffs, Eigen::ArrayBase<T0> &signal) const {
152  if (size() == 0) throw std::runtime_error("Empty wavelets: adjoint operation undefined");
153  if (signal.cols() == 1) {
154  if (coeffs.rows() % (1u << max_levels()) != 0)
155  throw std::length_error("Inconsistent number of columns and wavelet levels");
156  } else
157  SOPT_WAVELET_ERROR_MACRO(coeffs);
158  if (coeffs.cols() % size() != 0)
159  throw std::length_error(
160  "Columns of coefficient matrix and number of wavelets are inconsistent");
161  if (coeffs.rows() != signal.rows() or coeffs.cols() != signal.cols() * static_cast<t_int>(size()))
162  signal.derived().resize(coeffs.rows(), coeffs.cols() / size());
163  if (coeffs.rows() != signal.rows() or coeffs.cols() != signal.cols() * static_cast<t_int>(size()))
164  throw std::length_error("Incorrect size for output matrix(or could not resize)");
165  auto const Ncols = signal.cols();
166  {
167 #ifndef SOPT_OPENMP
168  SOPT_TRACE("Calling indirect sara without threads");
169 #endif
170  signal = T0::Zero(signal.rows(), signal.cols());
171  if (Ncols == 1)
172  for (size_type i = 0; i < size(); ++i) signal.col(0) += at(i).indirect(coeffs.col(i));
173  else
174  for (size_type i = 0; i < size(); ++i)
175  signal += at(i).indirect(coeffs.leftCols((i + 1) * Ncols).rightCols(Ncols));
176  }
177  signal /= std::sqrt(size());
178 }

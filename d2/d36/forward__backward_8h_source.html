223 void ForwardBackward<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual) const {
224  g_proximal(out, mu() * beta(), out - beta() * residual / std::pow(sigma(), 2));
225  PhiTPhi_(residual, out);
226  residual = residual - target();

229 template <class SCALAR>
230 typename ForwardBackward<SCALAR>::Diagnostic ForwardBackward<SCALAR>::
231 operator()(t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
232 
233  SOPT_HIGH_LOG("Performing Forward-Backward");
234  sanity_check(x_guess, res_guess);
235 
236  t_Vector residual = res_guess;
237  out = x_guess;
238 
239  t_uint niters(0);
240  bool converged = false;
241  for(; (not converged) && (niters < itermax()); ++niters) {
242  SOPT_LOW_LOG(" - [FB] Iteration {}/{}", niters, itermax());
243  iteration_step(out, residual);
244  SOPT_LOW_LOG(" - [FB] Sum of residuals: {}", residual.array().abs().sum());
245  SOPT_LOW_LOG(" - [FB] l2 objective function: {}",
246  0.5 * std::pow(sopt::l2_norm(residual) / sigma(), 2));
247  converged = is_converged(out, residual);
248  }
249 
250  if(converged) {
251  SOPT_MEDIUM_LOG(" - [FB] converged in {} of {} iterations", niters, itermax());
252  } else if(static_cast<bool>(is_converged())) {
253  // not meaningful if not convergence function
254  SOPT_ERROR(" - [FB] did not converge within {} iterations", itermax());
255  }
256  return {niters, converged, std::move(residual)};
257 }

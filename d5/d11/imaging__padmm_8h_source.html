275 template <class SCALAR>
276 typename ImagingProximalADMM<SCALAR>::Diagnostic ImagingProximalADMM<SCALAR>::
277 operator()(t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
278  SOPT_HIGH_LOG("Performing Proximal ADMM with L1 and L2 operators");
279  // The f proximal is an L1 proximal that stores some diagnostic result
280  Diagnostic result;
281  auto const f_proximal = [this, &result](t_Vector &out, Real gamma, t_Vector const &x) {
282  result.l1_diagnostic = this->l1_proximal(out, gamma, x);
283  };
284  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
285  this->l2ball_proximal()(out, gamma, x);
286  };
287  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
288  "Objective function");
289  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
290  return this->is_converged(scalvar, x, residual);
291  };
292  auto const padmm = PADMM(f_proximal, g_proximal, target())
293  .itermax(itermax())
294  .gamma(gamma())
295  .nu(nu())
296  .lagrange_update_scale(lagrange_update_scale())
297  .Phi(Phi())
298  .is_converged(convergence);
299  static_cast<typename PADMM::Diagnostic &>(result) = padmm(out, std::tie(guess, res));
300  return result;
301 }

303 template <class SCALAR>
304 bool ImagingProximalADMM<SCALAR>::residual_convergence(t_Vector const &x,
305  t_Vector const &residual) const {
306  if(static_cast<bool>(residual_convergence()))
307  return residual_convergence()(x, residual);
308  if(residual_tolerance() <= 0e0)
309  return true;
310  auto const residual_norm = sopt::l2_norm(residual, l2ball_proximal_weights());
311  SOPT_LOW_LOG(" - [PADMM] Residuals: {} <? {}", residual_norm, residual_tolerance());
312  return residual_norm < residual_tolerance();

317  t_Vector const &x,

327 template <class SCALAR>
328 bool ImagingProximalADMM<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
329  t_Vector const &x, t_Vector const &residual) const {
330  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
331  auto const res = residual_convergence(x, residual);
332  auto const obj = objective_convergence(scalvar, x, residual);
333  // beware of short-circuiting!
334  // better evaluate each convergence function everytime, especially with mpi
335  return user and res and obj;

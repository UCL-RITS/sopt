15 template <class T> sopt::Matrix<T> concatenated_permutations(sopt::t_uint i, sopt::t_uint j) {
16  std::vector<size_t> cols(j);
17  std::iota(cols.begin(), cols.end(), 0);
18  std::shuffle(cols.begin(), cols.end(), rengine);
19 
20  assert(j % i == 0);
21  auto const N = j / i;
22  auto const elem = 1e0 / std::sqrt(static_cast<typename sopt::real_type<T>::type>(N));
23  sopt::Matrix<T> result = sopt::Matrix<T>::Zero(i, cols.size());

25  result(cols[k] / N, k) = elem;

16 t_real power_method(const sopt::LinearTransform<T> &op, const t_uint &niters,
17  const t_real &relative_difference, const T &initial_vector) {
18  /*
19  Attempt at coding the power method, returns thesqrt of the largest eigen value of a linear
20  operator composed with its adjoint niters:: max number of iterations relative_difference::
21  percentage difference at which eigen value has converged
22  */
23  if (niters <= 0) return 1;
24  t_real estimate_eigen_value = 1;
25  t_real old_value = 0;
26  T estimate_eigen_vector = initial_vector;
27  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_vector.matrix().stableNorm();
28  SOPT_DEBUG("Starting power method");
29  SOPT_DEBUG(" -[PM] Iteration: 0, norm = {}", estimate_eigen_value);
30  for (t_int i = 0; i < niters; ++i) {
31  estimate_eigen_vector = op.adjoint() * (op * estimate_eigen_vector);
32  estimate_eigen_value = estimate_eigen_vector.matrix().stableNorm();
33  if (estimate_eigen_value <= 0) throw std::runtime_error("Error in operator.");
34  if (estimate_eigen_value != estimate_eigen_value)
35  throw std::runtime_error("Error in operator or data corrupted.");
36  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_value;
37  t_real const rel_diff = std::abs(old_value - estimate_eigen_value) / old_value;
38  SOPT_DEBUG(" -[PM] Iteration: {}, norm = {}", i + 1, estimate_eigen_value);
39  SOPT_DEBUG(" -[PM] Relative Difference = {} ( < {})", std::sqrt(rel_diff), relative_difference);
40  if (relative_difference * relative_difference > rel_diff) {
41  old_value = estimate_eigen_value;
42  SOPT_DEBUG("Converged to norm = {}, relative difference < {}", std::sqrt(old_value),
43  relative_difference);
44  break;
45  }
46  old_value = estimate_eigen_value;
47  }
48  return std::sqrt(old_value);
49 }

146  auto const rel_val = std::abs((magnitude - previous_magnitude) / previous_magnitude);
147  converged = rel_val < tolerance();
148  SOPT_INFO(" - [PM] iteration {}/{} -- norm: {}", niters, itermax(), magnitude);
149 
150  eigenvector /= magnitude;
151  previous_magnitude = magnitude;
152  }
153  // check function exists, otherwise, don't know if convergence is meaningful
154  if (not converged) {
155  SOPT_WARN(" - [PM] did not converge within {} iterations", itermax());
156  } else {
157  SOPT_INFO(" - [PM] converged in {} of {} iterations", niters, itermax());
158  }
159  return DiagnosticAndResult{itermax(), converged, previous_magnitude, eigenvector.normalized()};

195 template <class ALGORITHM>
196 typename Reweighted<ALGORITHM>::ReweightedResult Reweighted<ALGORITHM>::
197 operator()(ReweightedResult const &warm) const {
198  SOPT_HIGH_LOG("Starting reweighted scheme");
199  // Copies inner algorithm, so that operator() can be constant
200  Algorithm algo(algorithm());
201  ReweightedResult result(warm);
202 
203  auto delta = std::max(standard_deviation(reweightee(warm.algo.x)), min_delta());
204  SOPT_LOW_LOG("- Initial delta: {}", delta);
205  for(result.niters = 0; result.niters < itermax(); ++result.niters) {
206  SOPT_LOW_LOG("Reweigting iteration {}/{} ", result.niters, itermax());
207  SOPT_LOW_LOG(" - delta: {}", delta);
208  result.weights = delta / (delta + reweightee(result.algo.x).array().abs());
209  set_weights(algo, result.weights);
210  result.algo = algo(result.algo);
211  if(is_converged(result.algo.x)) {
212  SOPT_MEDIUM_LOG("Reweighting scheme did converge in {} iterations", result.niters);
213  result.good = true;
214  break;
215  }
216  delta = std::max(min_delta(), update_delta(delta));
217  }
218  // result is always good if no convergence function is defined
219  if(not is_converged())
220  result.good = true;
221  else if(not result.good)
222  SOPT_ERROR("Reweighting scheme did *not* converge in {} iterations", itermax());
223  return result;
224 }

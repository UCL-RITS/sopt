204 LinearTransform<Vector<typename DERIVED::Scalar>>
205 linear_transform(Eigen::MatrixBase<DERIVED> const &A) {
206  details::MatrixToLinearTransform<Matrix<typename DERIVED::Scalar>> const matrix(A);
207  if(A.rows() == A.cols())
208  return {matrix, matrix.adjoint()};
209  else {
210  t_int const gcd = details::gcd(A.cols(), A.rows());
211  t_int const a = A.cols() / gcd;
212  t_int const b = A.rows() / gcd;
213  return {matrix, matrix.adjoint(), {{b, a, 0}}};
214  }

209 template <class DERIVED>
210 LinearTransform<Vector<typename DERIVED::Scalar>> linear_transform(
211  Eigen::MatrixBase<DERIVED> const &A) {
212  details::MatrixToLinearTransform<Matrix<typename DERIVED::Scalar>> const matrix(A);
213  if (A.rows() == A.cols())
214  return {matrix, matrix.adjoint()};
215  else {
216  t_int const gcd = details::gcd(A.cols(), A.rows());
217  t_int const a = A.cols() / gcd;
218  t_int const b = A.rows() / gcd;
219  return {matrix, matrix.adjoint(), {{b, a, 0}}};
220  }
221 }

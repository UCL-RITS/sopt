191 template <class SCALAR>
192 typename SDMM<SCALAR>::Diagnostic SDMM<SCALAR>::
193 operator()(t_Vector &out, t_Vector const &input) const {
194 
195  sanity_check(input);
196  bool convergence = false;
197  t_uint niters(0);
198  // Figures out where itermax or convergence reached
199  auto const has_finished = [&convergence, &niters, this](t_Vector const &out) {
200  convergence = is_converged(out);
201  return niters >= itermax() or convergence;
202  };
203 
204  SOPT_HIGH_LOG("Performing SDMM ");
205  out = input;
206  t_Vectors y(transforms().size()), z(transforms().size());
207 
208  // Initial step replaces iteration update with initialization
209  initialization(y, z, input);
210  auto cg_diagnostic = solve_for_xn(out, y, z);
211 
212  while(not has_finished(out)) {
213  SOPT_LOW_LOG("Iteration {}/{}. ", niters, itermax());
214  // computes y and z from out and transforms
215  update_directions(y, z, out);
216  SOPT_LOW_LOG(" - sum z_ij = {}",
217  std::accumulate(z.begin(), z.end(), Scalar(0e0),
218  [](Scalar const &a, t_Vector const &z) { return a + z.sum(); }));
219  // computes x = L^-1 y
220  cg_diagnostic = solve_for_xn(out, y, z);
221  SOPT_LOW_LOG(" - CG Residual = {} in {}/{} iterations", cg_diagnostic.residual,
222  cg_diagnostic.niters, conjugate_gradient().itermax());
223 
224  ++niters;
225  }
226  return {niters, convergence, cg_diagnostic};
227 }

229 template <class SCALAR>
230 ConjugateGradient::Diagnostic
231 SDMM<SCALAR>::solve_for_xn(t_Vector &out, t_Vectors const &y, t_Vectors const &z) const {
232 
233  assert(z.size() == transforms().size());
234  assert(y.size() == transforms().size());
235  SOPT_TRACE("Solving for x_n");
236 
237  // Initialize b of A x = b = sum_i L_i^H(z_i - y_i)
238  t_Vector b = out.Zero(out.size());
239  for(t_uint i(0); i < transforms().size(); ++i)
240  b += transforms(i).adjoint() * (y[i] - z[i]);
241  if(b.stableNorm() < 1e-12) {
242  out.fill(0e0);
243  return {0, 0, true};
244  }
245 
246  // Then create operator A
247  auto A = [this](t_Vector &out, t_Vector const &input) {
248  out = out.Zero(input.size());
249  for(auto const &transform : this->transforms())
250  out += transform.adjoint() * (transform * input).eval();
251  };
252 
253  // Call conjugate gradient
254  auto const diagnostic = this->conjugate_gradient(out, A, b);
255  if(not diagnostic.good) {
256  SOPT_ERROR("CG error - iterations: {}/{} - residuals {}\n", diagnostic.niters,
257  conjugate_gradient().itermax(), diagnostic.residual);
258  SOPT_THROW("Conjugate gradient failed to converge");
259  }
260 
261  return diagnostic;
262 }

286 template <class SCALAR> void SDMM<SCALAR>::sanity_check(t_Vector const &x) const {
287  bool doexit = false;
288  if(proximals().size() != transforms().size()) {
289  SOPT_ERROR("Internal error: number of proximals and transforms do not match");
290  doexit = true;
291  }
292  if(x.size() == 0)
293  SOPT_WARN("Input vector has zero size");
294  if(size() == 0)
295  SOPT_WARN("No operators - SDMM is empty");
296  for(t_uint i(0); i < size(); ++i) {
297  auto const xdual = t_Vector::Zero((transforms(i) * x).size());
298  auto const r = (transforms(i).adjoint() * xdual).size();
299  if(r != x.size()) {
300  SOPT_ERROR("Output size of transform {} and input do not match: {} vs {}", i, r, x.size());
301  doexit = true;
302  }
303  }
304  if(doexit)
305  SOPT_THROW("Input to SDMM is inconsistent");
306 }

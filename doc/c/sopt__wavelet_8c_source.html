127  SOPT_ERROR_MEM_ALLOC_CHECK(h1);

129  /* Lowpass and highpass analysis (decomposition) filters */
130  for (i=0; i<param_wav.h_size; i++)
131  {
132  *(h0+i) = *(param_wav.h+param_wav.h_size-i-1);
133  *(h1+i) = *(param_wav.h+i);

136  *(h1+i) = -*(h1+i);

139  for (decomp_level = 0; decomp_level<param_wav.nb_levels; decomp_level++ )
140  {
141  /* Transform along rows */
142  for (i=0; i<param_wav.nx1/pow(2, decomp_level); i++){
143  /* Copy input */
144  if(decomp_level==0){
145  for(j=0; j<param_wav.nx2/pow(2, decomp_level); j++)
146  *(in+j) = *(x+i+param_wav.nx1*j);
147  }

150  *(in+j) = *(w+i+param_wav.nx1*j);
151  }

154  sopt_wavelet_convd(out_l, out_h, in, h0, h1, param_wav.nx2/pow(2, decomp_level), param_wav.h_size);
155  /* Downsample and copy output */

159  }

161  {

163  *(w+i+param_wav.nx1*j) = *(out_h+2*(j-dim_max));
164  }
165  }
166

169  /* Copy input */
170  for(i=0; i<param_wav.nx1/pow(2, decomp_level); i++)
171  *(in+i) = *(w+i+param_wav.nx1*j);
172 
173  /* Perform filtering lowpass and highpass */
174  sopt_wavelet_convd(out_l, out_h, in, h0, h1, param_wav.nx1/pow(2, decomp_level), param_wav.h_size);
175  /* Downsample and copy output */

177  {

179  }

184  }

186  }
187  }

230  /* Lowpass and highpass synthesis (reconstruction) filters */
231  for (i=0; i<param_wav.h_size; i++)
232  {
233  *(h0+i) = *(param_wav.h+i);
234  *(h1+i) = *(param_wav.h+param_wav.h_size-i-1);
235  }
236  for (i=1; i<param_wav.h_size; i+=2)
237  *(h1+i) = -*(h1+i);
238

240  cblas_dcopy(param_wav.nx1*param_wav.nx2, w, 1, x, 1);
241 
242  /* Main loop */
243  for (decomp_level = param_wav.nb_levels; decomp_level>0; decomp_level--)
244  {
245  /* Transform along rows */
246  for (i=0; i<param_wav.nx1/pow(2, decomp_level-1); i++){
247

250  for (j=0; j<param_wav.nx2/pow(2, decomp_level); j++){
251

253  if((2*j+1)<param_wav.nx2)
254  *(in_l+2*j+1) = 0.0;
255  }
256  /* High frequency part */

259  *(in_h+2*(j-dim_max)) = *(x+i+param_wav.nx1*j);
260  if((2*(j-dim_max)+1)<param_wav.nx2)
261  *(in_h+2*(j-dim_max)+1) = 0.0;
262  }
263  /* Perform filtering lowpass and highpass */
264  sopt_wavelet_convu(out, in_l, in_h, h0, h1, param_wav.nx2/pow(2, decomp_level-1), param_wav.h_size);
265  /* Copy output */
266  for(j=0; j<param_wav.nx2/pow(2, decomp_level-1); j++)
267  *(x+i+param_wav.nx1*j) = *(out+j);
268  }
269 
270  /* Transform along columns */
271  for (j=0; j<param_wav.nx2/pow(2, decomp_level-1); j++){
272 
273  /* Copy inputs and upsample */
274  /* Low frequency part */
275  for(i=0; i<param_wav.nx1/pow(2, decomp_level); i++)
276  {
277  *(in_l+2*i) = *(x+i+param_wav.nx1*j);

293  *(x+i+param_wav.nx1*j) = *(out+i);

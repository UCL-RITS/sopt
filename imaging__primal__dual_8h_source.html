269 template <class SCALAR>
270 typename ImagingPrimalDual<SCALAR>::Diagnostic ImagingPrimalDual<SCALAR>::operator()(
271  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
272  SOPT_HIGH_LOG("Performing Primal Dual with L1 and L2 operators");
273  // The f proximal is an L1 proximal that stores some diagnostic result
274  auto const f_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
275  this->l1norm_proximal()(out, gamma, x);
276  };
277  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
278  this->l2ball_proximal()(out, gamma, x);
279  // applying preconditioning
280  for (t_int i = 0; i < this->precondition_iters(); i++)
281  this->l2ball_proximal()(
282  out, gamma,
283  out - this->precondition_stepsize() *
284  (out.array() * this->precondition_weights().array() - x.array()).matrix());
285 
286  if (this->precondition_iters() > 0) out = out.array() * this->precondition_weights().array();
287  };
288  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
289  "Objective function");
290  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
291  return this->is_converged(scalvar, x, residual);
292  };
293  const bool positive = positivity_constraint();
294  const bool real = real_constraint();
295  t_Constraint constraint = [real, positive](t_Vector &out, const t_Vector &x) {
296  if (real) out.real() = x.real();
297  if (positive) out = sopt::positive_quadrant(x);
298  };
299  auto const pd = PD(f_proximal, g_proximal, target())
300  .itermax(itermax())
301  .constraint(constraint)
302  .sigma(sigma())
303  .tau(tau())
304  .gamma(gamma())
305  .update_scale(update_scale())
306  .xi(xi())
307  .rho(rho())
308  .nu(nu())
309  .gamma(gamma())
310  .Phi(Phi())
311  .Psi(Psi())
312  .is_converged(convergence);
313  Diagnostic result;
314  static_cast<typename PD::Diagnostic &>(result) = pd(out, std::tie(guess, res));
315  return result;
316 }

318 template <class SCALAR>
319 bool ImagingPrimalDual<SCALAR>::residual_convergence(t_Vector const &x,
320  t_Vector const &residual) const {
321  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
322  if (residual_tolerance() <= 0e0) return true;
323  auto const residual_norm = sopt::l2_norm(residual, l2ball_proximal_weights());
324  SOPT_LOW_LOG(" - [Primal Dual] Residuals: {} <? {}", residual_norm, residual_tolerance());
325  return residual_norm < residual_tolerance();

338 template <class SCALAR>
339 bool ImagingPrimalDual<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
340  t_Vector const &x, t_Vector const &residual) const {
341  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
342  auto const res = residual_convergence(x, residual);
343  auto const obj = objective_convergence(scalvar, x, residual);
344  // beware of short-circuiting!
345  // better evaluate each convergence function everytime, especially with mpi
346  return user and res and obj;
347 }

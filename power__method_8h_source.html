19 std::tuple<t_real, T> power_method(const sopt::LinearTransform<T> &op, const t_uint niters,
20  const t_real relative_difference, const T &initial_vector) {
21  /*
22  Attempt at coding the power method, returns the sqrt of the largest eigen value of a linear
23  operator composed with its adjoint niters:: max number of iterations relative_difference::
24  percentage difference at which eigen value has converged
25  */
26  if (niters <= 0) return std::make_tuple(1., initial_vector);
27  t_real estimate_eigen_value = 1;
28  t_real old_value = 0;
29  T estimate_eigen_vector = initial_vector;
30  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_vector.matrix().stableNorm();
31  SOPT_DEBUG("Starting power method");
32  SOPT_DEBUG(" -[PM] Iteration: 0, norm = {}", estimate_eigen_value);
33  bool converged = false;
34  ScalarRelativeVariation<t_real> scalvar(relative_difference, 0., "Eigenvalue");
35  for (t_int i = 0; i < niters; ++i) {
36  estimate_eigen_vector = op.adjoint() * (op * estimate_eigen_vector).eval();
37  estimate_eigen_value = estimate_eigen_vector.matrix().stableNorm();
38  if (estimate_eigen_value <= 0) throw std::runtime_error("Error in operator.");
39  if (estimate_eigen_value != estimate_eigen_value)
40  throw std::runtime_error("Error in operator or data corrupted.");
41  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_value;
42  SOPT_DEBUG(" -[PM] Iteration: {}, norm = {}", i + 1, estimate_eigen_value);
43  converged = scalvar(std::sqrt(estimate_eigen_value));
44  old_value = estimate_eigen_value;
45  if (converged) {
46  SOPT_DEBUG("Converged to norm = {}, relative difference < {}", std::sqrt(old_value),
47  relative_difference);
48  break;
49  }
50  }
51  return std::make_tuple(std::sqrt(old_value), estimate_eigen_vector);
52 }

176
